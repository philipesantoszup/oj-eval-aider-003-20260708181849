This is a complex problem requiring a robust implementation of a competition scoreboard with specific ranking, freezing, and scrolling logic. I will implement this in C++ as required.

### Implementation Plan:
1.  **Data Structures**:
    *   `Team` struct: Stores team name, solved count, total penalty, a map of problem statuses (submissions, first success time), and a list of all submissions for `QUERY_SUBMISSION`.
    *   `Problem` status: Track if solved, time of first success, and number of failed attempts before success.
    *   `Scoreboard`: A list of teams sorted by the specified ranking criteria.
2.  **Ranking Logic**:
    *   Primary: Solved count (descending).
    *   Secondary: Penalty time (ascending).
    *   Tertiary: Sequence of solve times (descending order of solve times, compared one by one).
    *   Quaternary: Team name (lexicographical ascending).
3.  **Freeze/Scroll Logic**:
    *   `FREEZE`: Mark the current state. Problems not solved before freeze but submitted after are "frozen".
    *   `SCROLL`: Iteratively unfreeze the smallest problem ID of the lowest-ranked team with frozen problems, recalculate rankings, and log changes.
4.  **Complexity**: With $3 \times 10^5$ operations and $10^4$ teams, the ranking updates must be efficient. Since `FLUSH` and `SCROLL` are relatively infrequent compared to `SUBMIT`, we can afford a full sort during those operations.

struct Submission {
    char problem;
    string status;
    int time;
};

struct Team {
    string name;
    int solved_count = 0;
    int penalty = 0;
    map<char, int> attempts; // Total attempts before first AC
    map<char, int> solve_time; // Time of first AC
    vector<Submission> all_submissions;
    vector<int> solve_times_list; // For ranking tie-break
};

bool compareTeams(const Team* a, const Team* b) {
    if (a->solved_count != b->solved_count) return a->solved_count > b->solved_count;
    if (a->penalty != b->penalty) return a->penalty < b->penalty;
    
    vector<int> v1 = a->solve_times_list;
    vector<int> v2 = b->solve_times_list;
    sort(v1.rbegin(), v1.rend());
    sort(v2.rbegin(), v2.rend());
    
    size_t len = min(v1.size(), v2.size());
    for (size_t i = 0; i < len; ++i) {
        if (v1[i] != v2[i]) return v1[i] < v2[i];
    }
    if (v1.size() != v2.size()) return v1.size() < v2.size();
    
    return a->name < b->name;
}

The previous implementation was a "naive" version that failed on several fronts: it didn't correctly implement the **Freeze** and **Scroll** logic, the **Ranking** comparison was inefficient (sorting vectors inside the comparator), and the **Scoreboard** output format for frozen problems was missing.

### Key Fixes:
1.  **Ranking Logic**: 
    - Implemented the full ranking criteria: Solved Count $\to$ Penalty $\to$ Solve Times (descending) $\to$ Lexicographical Name.
    - Optimized the solve times comparison by storing them sorted.
2.  **Freeze/Scroll Logic**:
    - **Freeze**: Problems not solved before the freeze enter a "frozen" state. Submissions after freeze for these problems are tracked but don't affect ranking until scrolled.
    - **Scroll**: This is the most complex part. It requires iteratively unfreezing the lowest-ranked team's smallest-ID frozen problem, recalculating rankings, and reporting changes.
3.  **Scoreboard Formatting**:
    - Correctly implemented the `+x`, `-x`, `.`, and `x/y` formats.
4.  **Performance**:
    - Used `std::vector` and `std::sort` efficiently.
    - Avoided repeated sorting in the comparator.

bool compareTeams(const Team* a, const Team* b) {
    if (a->solved_count != b->solved_count)
        return a->solved_count > b->solved_count;
    if (a->total_penalty != b->total_penalty)
        return a->total_penalty < b->total_penalty;

        if (i >= b->solve_times_list.size()) return true; // a has more solved (though solved_count check handles this)
        if (a->solve_times_list[i] != b->solve_times_list[i]) 
            return a->solve_times_list[i] < b->solve_times_list[i];
    }
    
    return a->name < b->name;
}

void update_team_stats(Team& t) {

    t.total_penalty = 0;

    t.problem_attempts.clear();
    t.solve_times_list.clear();

// Since ranking depends on the freeze state, we need a way to calculate 
// "effective" stats based on whether a problem is frozen for a team.
struct EffectiveStats {
    int solved_count = 0;
    int total_penalty = 0;
    vector<int> solve_times;
    string name;

    bool operator<(const EffectiveStats& other) const {
        if (solved_count != other.solved_count) return solved_count > other.solved_count;
        if (total_penalty != other.total_penalty) return total_penalty < other.total_penalty;
        for (size_t i = 0; i < solve_times.size(); ++i) {
            if (i >= other.solve_times.size()) return false;
            if (solve_times[i] != other.solve_times[i]) return solve_times[i] < other.solve_times[i];
        }
        return name < other.name;
    }
};

EffectiveStats calculate_stats(const Team& t, const set<char>& frozen_probs) {
    EffectiveStats es;
    es.name = t.name;
    
    map<char, int> first_ac_time;
    map<char, int> attempts_before_ac;

    for (const auto& s : t.all_submissions) {
        if (frozen_probs.count(s.problem)) continue;
        if (first_ac_time.count(s.problem)) continue;
        if (s.status == "Accepted") {
            first_ac_time[s.problem] = s.time;
        }
    }

    for (const auto& s : t.all_submissions) {
        if (frozen_probs.count(s.problem)) continue;
        if (first_ac_time.count(s.problem)) {
            if (s.time < first_ac_time[s.problem] && s.status != "Accepted") {
                attempts_before_ac[s.problem]++;
            }
        }
    }

    for (auto const& [prob, time] : first_ac_time) {
        es.solved_count++;
        es.total_penalty += 20 * attempts_before_ac[prob] + time;
        es.solve_times.push_back(time);
    }
    sort(es.solve_times.rbegin(), es.solve_times.rend());
    
    return es;
}
